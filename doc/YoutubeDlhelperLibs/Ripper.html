def self.rip(filename)
  puts 'Initializing the file'
  movie = FFMPEG::Movie.new("#{filename}.m4a")
  puts 'Initializing finished'
  movie.valid?
  puts 'Validated'
  movie.transcode("#{filename}.m4a", "-acodec libmp3lame -ac 2 -ab 192k #{filename}.mp3")
  puts 'Transcoded'
end
